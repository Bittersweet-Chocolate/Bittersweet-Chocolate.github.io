#### 冒泡排序

冒泡排序：时间复杂度 O(n²)

起手式

```
var bubbleSort = function (arr) {
    var i = 0;
    for (; i < arr.length; i++) {
        for (var j = 0; j < arr.length - 1 - i; j++) {
            if (arr[j] > arr[j + 1])
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        }
    }
    return arr
}

```

#### 插入排序

时间复杂度：O(N^(1-2))

```
var insertSort = function (arr) {
    for (var i = 1; i < arr.length; i++) {
        for (var j = i - 1; j >= 0 && arr[j] >= arr[j + 1]; j--) {
            [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
        }
    }
    return arr
}
```

#### 快速排序

时间复杂度：O (nlogn)

###### 第一种：简单明的概念

```
var quickSort = function (arr) {
    if (arr.length <= 1) return arr
    let left = [],
        right = [];
    const idx = Math.floor(arr.length / 2)
    const val = arr.splice(idx,1)[0]
    for (var i = 0; i < arr.length; i++) {
        if (arr[i] >= val) right.push(arr[i])
        else left.push(arr[i])
    }
    return quickSort(left).concat([val], quickSort(right))
}
```

###### 第二种：速度更快

```
var quickSort = function (arr) {
    var sort = (arr, left = 0, right = arr.length - 1) => {
        if (left >= right) return
        let l = left,
            r = right;
        const val = arr[right]
        while (l < r) {
            while (l < r && arr[l] <= val) {
                l++
            }
            arr[r] = arr[l]
            while (r > l && arr[r] >= val) {
                r--
            }
            arr[l] = arr[r]
        }
        arr[r] = val
        sort(arr,left,r-1);
        sort(arr,r+1,right);
    }
    var newArr = arr.concat()
    sort(newArr)
    return newArr
}
```

#### 二分法查询

对已经排好序的数组查找是否存在对应数据

```
var binarySearch = function (arr, item) {
    let left = 0,
        right = arr.length - 1;
    let idx = Math.floor((left + right) / 2)
    while (left < right) {
        if (arr[idx] === item) return idx
        else if (arr[idx] > item) right = idx - 1
        else left = idx + 1
    }
    return -1
}
```
